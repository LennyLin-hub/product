function auto_schedule(tasks, resources, calendars, compat, params, changeover_rule):
    # 1) 只拿可排任务：状态 READY 且前置已排/可推算 earliest_start
    ready_tasks = filter(tasks, task.status == READY)

    # 2) 排序：优先级 priority DESC，再交期 due_date ASC（或纯 EDD）
    ready_tasks.sort(key = (task.order_priority DESC, task.due_date ASC, task.std_duration_min ASC))

    for task in ready_tasks:
        if task.op_code in [SETUP, INJECT]:
            candidates = []
            for machine in machines_that_can_do(task.product_id):
                for mold in molds_that_match(task.product_id):
                    if not is_compatible(machine, mold): continue
                    if mold.status not in [AVAILABLE, BUSY]: continue  # MAINTENANCE/REPAIR 直接不可用
                    # 3) 计算该组合的最早开始时间：同时满足
                    # - 任务 earliest_start
                    # - machine 可用时段（含已占用的 assignment）
                    # - mold 可用时段（含已占用的 assignment）
                    # - 班次日历可用
                    t0 = max(task.earliest_start,
                             next_free_time(machine),
                             next_free_time(mold),
                             next_calendar_open(machine.calendar_id, after=max(...)))

                    # 4) SETUP 需要换型时间（来自 machine 上一个任务）
                    if task.op_code == SETUP:
                        prev = last_task_on(machine)
                        chg = calc_changeover(prev, task, changeover_rule)  # same_mold=0 etc.
                        duration = task.std_duration_min + chg
                    else:
                        duration = task.std_duration_min

                    t1 = add_minutes_with_calendar(t0, duration, machine.calendar_id)

                    candidates.append((t0, t1, machine, mold, duration))

            # 5) 选最早完工（或最早开始）的一组
            best = argmin(candidates, key = (t1, t0))
            commit_assignment(task, best.machine, best.mold, best.t0, best.t1, best.duration)
            emit_event(task, DISPATCHED, time=now(), resource_id=best.machine)

        else if task.op_code == POST_QC_PUTAWAY:
            ws = pick_earliest_free(workstations_that_can_do(task.product_id))
            t0 = max(task.earliest_start, next_free_time(ws), next_calendar_open(ws.calendar_id, after=...))
            t1 = add_minutes_with_calendar(t0, task.std_duration_min, ws.calendar_id)
            commit_assignment(task, workstation_id=ws, start=t0, end=t1, duration=task.std_duration_min)
            emit_event(task, DISPATCHED, time=now(), resource_id=ws)

    return all_assignments
